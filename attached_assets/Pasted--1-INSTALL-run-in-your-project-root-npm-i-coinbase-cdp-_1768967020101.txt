# 1) INSTALL (run in your project root)
npm i @coinbase/cdp-react @coinbase/cdp-core @coinbase/cdp-hooks

# 2) .env (create at project root)
VITE_CDP_PROJECT_ID=PASTE_YOUR_CDP_PROJECT_ID_HERE

# 3) src/main.tsx (PASTE THIS WHOLE FILE)
import React from "react";
import ReactDOM from "react-dom/client";
import { CDPReactProvider, type Config } from "@coinbase/cdp-react";
import { AuthButton } from "@coinbase/cdp-react/components/AuthButton";
import { useIsInitialized, useIsSignedIn } from "@coinbase/cdp-hooks";
import "./index.css";

const config: Config = {
  projectId: import.meta.env.VITE_CDP_PROJECT_ID,
  appName: "DJZS",
  // Creates a Base smart wallet on login (gasless-ready pattern; paymaster is separate)
  ethereum: { createOnLogin: "smart" },
  // Auth methods (AuthButton will render what you’ve enabled in CDP)
  authMethods: ["email", "oauth:google", "oauth:apple"],
};

function App() {
  const { isInitialized } = useIsInitialized();
  const { isSignedIn } = useIsSignedIn();

  if (!isInitialized) return <div style={{ padding: 24 }}>Loading…</div>;

  return (
    <div style={{ minHeight: "100vh", display: "grid", placeItems: "center" }}>
      <div style={{ width: 520, maxWidth: "92vw" }}>
        <h1 style={{ fontSize: 28, marginBottom: 8 }}>DJZS</h1>
        <p style={{ opacity: 0.8, marginBottom: 18 }}>
          Sign in — a wallet is created automatically.
        </p>

        {isSignedIn ? (
          <div
            style={{
              padding: 16,
              borderRadius: 14,
              border: "1px solid rgba(255,255,255,0.12)",
            }}
          >
            Signed in ✅
          </div>
        ) : (
          <div
            style={{
              padding: 16,
              borderRadius: 14,
              border: "1px solid rgba(255,255,255,0.12)",
            }}
          >
            <AuthButton />
            <div style={{ marginTop: 10, fontSize: 12, opacity: 0.75 }}>
              If this doesn’t load on dj-z-s.box, the #1 fix is to allowlist
              <b> https://dj-z-s.box </b> in your CDP Portal “Domains”.
            </div>
          </div>
        )}
      </div>
    </div>
  );
}

ReactDOM.createRoot(document.getElementById("root")!).render(
  <React.StrictMode>
    <CDPReactProvider config={config}>
      <App />
    </CDPReactProvider>
  </React.StrictMode>
);
