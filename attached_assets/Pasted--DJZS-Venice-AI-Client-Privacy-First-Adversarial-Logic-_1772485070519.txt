// ============================================================================
// VENICE API CLIENT
// ============================================================================

interface VeniceRequestOptions {
  model?: VeniceModel;
  maxTokens?: number;
  temperature?: number;
  stream?: boolean;
  jsonSchema?: object;
  includeVeniceSystemPrompt?: boolean;
}

interface VeniceMessage {
  role: "system" | "user" | "assistant";
  content: string;
}

interface VeniceResponse {
  id: string;
  object: string;
  created: number;
  model: string;
  choices: {
    index: number;
    message: {
      role: string;
      content: string;
    };
    finish_reason: string;
  }[];
  usage: {
    prompt_tokens: number;
    completion_tokens: number;
    total_tokens: number;
  };
}

export class VeniceClient {
  private apiKey: string;
  private baseUrl: string;

  constructor(apiKey?: string, baseUrl?: string) {
    this.apiKey = apiKey || VENICE_API_KEY || "";
    this.baseUrl = baseUrl || VENICE_BASE_URL;

    if (!this.apiKey) {
      throw new Error("VENICE_API_KEY is required");
    }
  }

  /**
   * Raw chat completion request to Venice API
   */
  async chat(
    messages: VeniceMessage[],
    options: VeniceRequestOptions = {}
  ): Promise<VeniceResponse> {
    const {
      model = DEFAULT_MODEL,
      maxTokens = 2048,
      temperature = 0.3,
      stream = false,
      jsonSchema,
      includeVeniceSystemPrompt = false,
    } = options;

    const body: Record<string, unknown> = {
      model,
      messages,
      max_tokens: maxTokens,
      temperature,
      stream,
      venice_parameters: {
        include_venice_system_prompt: includeVeniceSystemPrompt,
      },
    };

    // Request structured JSON output if schema provided
    if (jsonSchema) {
      body.response_format = {
        type: "json_schema",
        json_schema: {
          name: "audit_response",
          strict: true,
          schema: jsonSchema,
        },
      };
    }

    const response = await fetch(`${this.baseUrl}/chat/completions`, {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        "Authorization": `Bearer ${this.apiKey}`,
      },
      body: JSON.stringify(body),
    });

    if (!response.ok) {
      const errorText = await response.text();
      throw new Error(`Venice API error ${response.status}: ${errorText}`);
    }

    return response.json();
  }

  /**
   * Streaming chat completion
   */
  async *chatStream(
    messages: VeniceMessage[],
    options: VeniceRequestOptions = {}
  ): AsyncGenerator<string, void, unknown> {
    const {
      model = DEFAULT_MODEL,
      maxTokens = 2048,
      temperature = 0.3,
      includeVeniceSystemPrompt = false,
    } = options;

    const response = await fetch(`${this.baseUrl}/chat/completions`, {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        "Authorization": `Bearer ${this.apiKey}`,
      },
      body: JSON.stringify({
        model,
        messages,
        max_tokens: maxTokens,
        temperature,
        stream: true,
        venice_parameters: {
          include_venice_system_prompt: includeVeniceSystemPrompt,
        },
      }),
    });

    if (!response.ok) {
      const errorText = await response.text();
      throw new Error(`Venice API error ${response.status}: ${errorText}`);
    }

    const reader = response.body?.getReader();
    if (!reader) throw new Error("No response body");

    const decoder = new TextDecoder();
    let buffer = "";

    while (true) {
      const { done, value } = await reader.read();
      if (done) break;

      buffer += decoder.decode(value, { stream: true });
      const lines = buffer.split("\n");
      buffer = lines.pop() || "";

      for (const line of lines) {
        if (line.startsWith("data: ")) {
          const data = line.slice(6);
          if (data === "[DONE]") return;

          try {
            const parsed = JSON.parse(data);
            const content = parsed.choices?.[0]?.delta?.content;
            if (content) yield content;
          } catch {
            // Skip malformed chunks
          }
        }
      }
    }
  }
}

export interface AuditRequest {
  strategyMemo: string;
  auditType?: string;
  targetSystem?: string;
  tier?: AuditTier;
  persona?: AdversarialPersona;
}

export interface StructuredAuditResult {
  verdict: "PASS" | "FAIL";
  risk_score: number;
  confidence: number;
  primary_bias_detected: string;
  flags: DJZSFlag[];
  logic_flaws: string[];
  scenarios?: {
    name: string;
    outcome: string;
    estimated_impact: string;
  }[];
  structural_recommendations: string[];
  action_items: string[];
  raw_response?: string;
  model_used: string;
  persona_used: AdversarialPersona;
}

/**
 * Run adversarial audit with structured output
 */
export async function runAdversarialAudit(
  client: VeniceClient,
  request: AuditRequest
): Promise<StructuredAuditResult> {
  const {
    strategyMemo,
    auditType = "general",
    targetSystem = "Unknown",
    tier = "micro",
    persona = "general",
  } = request;

  const systemPrompt = PERSONA_SYSTEM_PROMPTS[persona];
  const model = PERSONA_MODELS[persona];

  const userPrompt = `
AUDIT REQUEST
=============
Tier: ${tier}
Type: ${auditType}
Target: ${targetSystem}

STRATEGY MEMO
=============
${strategyMemo}

Analyze this memo for logical flaws, biases, and risks.
Return your analysis as a JSON object with these fields:
- verdict: "PASS" or "FAIL"
- risk_score: 0-100 (0 = flawless, 100 = critical failure)
- confidence: 0-1 (your confidence in this assessment)
- primary_bias_detected: string description or "None"
- flags: array of DJZS-LF failure codes with severity and description
- logic_flaws: array of specific flaw descriptions
- scenarios: array of stress-test scenarios (if applicable)
- structural_recommendations: array of improvement suggestions
- action_items: array of concrete next steps

Return ONLY the JSON object. No markdown, no explanation.
`.trim();

  const messages: VeniceMessage[] = [
    { role: "system", content: systemPrompt },
    { role: "user", content: userPrompt },
  ];

  try {
    const response = await client.chat(messages, {
      model,
      maxTokens: 2048,
      temperature: 0.2, // Low temperature for deterministic audits
      jsonSchema: AUDIT_RESPONSE_SCHEMA,
      includeVeniceSystemPrompt: false,
    });

    const content = response.choices[0]?.message?.content || "";
    
    // Parse JSON response (handle potential markdown wrapping)
    let parsed: StructuredAuditResult;
    try {
      const cleanContent = content
        .replace(/```json\n?/g, "")
        .replace(/```\n?/g, "")
        .trim();
      parsed = JSON.parse(cleanContent);
    } catch (parseError) {
      // If JSON parsing fails, return a structured error response
      console.error("Failed to parse Venice response as JSON:", content);
      parsed = {
        verdict: "FAIL",
        risk_score: 50,
        confidence: 0.3,
        primary_bias_detected: "Response parsing error",
        flags: [{
          code: "DJZS-X99",
          category: "Execution",
          name: "AUDIT_PARSE_ERROR",
          severity: "HIGH",
          description: "Failed to parse AI response as structured JSON",
          recommendation: "Retry audit or check model output format"
        }],
        logic_flaws: ["AI response was not valid JSON"],
        structural_recommendations: ["Retry with different model"],
        action_items: ["Review raw response for partial insights"],
        raw_response: content,
        model_used: model,
        persona_used: persona,
      };
      return parsed;
    }

    // Enrich with metadata
    return {
      ...parsed,
      model_used: model,
      persona_used: persona,
      raw_response: content,
    };

  } catch (error) {
    console.error("Venice API call failed:", error);
    throw error;
  }
}

/**
 * Run multi-persona audit (comprehensive analysis)
 */
export async function runMultiPersonaAudit(
  client: VeniceClient,
  request: AuditRequest
): Promise<{
  combined_verdict: "PASS" | "FAIL";
  combined_risk_score: number;
  persona_results: Record<AdversarialPersona, StructuredAuditResult>;
}> {
  const personas: AdversarialPersona[] = [
    "logic_auditor",
    "risk_hunter",
    "backtest_skeptic",
  ];

  // Run all persona audits in parallel
  const results = await Promise.all(
    personas.map(persona =>
      runAdversarialAudit(client, { ...request, persona })
    )
  );

  // Combine results
  const personaResults: Record<string, StructuredAuditResult> = {};
  let totalRiskScore = 0;
  let anyFail = false;

  personas.forEach((persona, index) => {
    personaResults[persona] = results[index];
    totalRiskScore += results[index].risk_score;
    if (results[index].verdict === "FAIL") anyFail = true;
  });

  return {
    combined_verdict: anyFail ? "FAIL" : "PASS",
    combined_risk_score: Math.round(totalRiskScore / personas.length),
    persona_results: personaResults as Record<AdversarialPersona, StructuredAuditResult>,
  };
}

// ============================================================================
// SIMPLE TEXT GENERATION (for non-audit use cases)
// ============================================================================

export async function generateText(
  client: VeniceClient,
  prompt: string,
  options: VeniceRequestOptions = {}
): Promise<string> {
  const messages: VeniceMessage[] = [
    { role: "user", content: prompt },
  ];

  const response = await client.chat(messages, options);
  return response.choices[0]?.message?.content || "";
}
