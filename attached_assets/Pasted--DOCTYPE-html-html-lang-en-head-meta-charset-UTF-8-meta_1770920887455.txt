    <script>
        class MusicManager {
            constructor() {
                this.db = null;
                this.currentTrack = null;
                this.audio = new Audio();
                this.isPlaying = false;
                this.currentZone = 'journaling';
                this.initializeDB();
                this.setupEventListeners();
                this.setupAudioEvents();
            }

            async initializeDB() {
                this.db = await idb.openDB('DJZSMusicDB', 1, {
                    upgrade(db) {
                        db.createObjectStore('tracks', { keyPath: 'id' });
                        db.createObjectStore('playlists', { keyPath: 'id' });
                    }
                });
                this.loadMusicList();
                this.loadPlaylists();
            }

            setupEventListeners() {
                const uploadArea = document.getElementById('uploadArea');
                const fileInput = document.getElementById('fileInput');
                const playBtn = document.getElementById('playBtn');
                const progressBar = document.getElementById('progressBar');
                const volumeSlider = document.getElementById('volumeSlider');

                uploadArea.addEventListener('click', () => fileInput.click());
                fileInput.addEventListener('change', (e) => this.handleFileUpload(e.target.files));

                uploadArea.addEventListener('dragover', (e) => {
                    e.preventDefault();
                    uploadArea.classList.add('dragover');
                });

                uploadArea.addEventListener('dragleave', () => {
                    uploadArea.classList.remove('dragover');
                });

                uploadArea.addEventListener('drop', (e) => {
                    e.preventDefault();
                    uploadArea.classList.remove('dragover');
                    this.handleFileUpload(e.dataTransfer.files);
                });

                playBtn.addEventListener('click', () => this.togglePlayPause());
                
                progressBar.addEventListener('click', (e) => {
                    const rect = progressBar.getBoundingClientRect();
                    const percent = (e.clientX - rect.left) / rect.width;
                    this.audio.currentTime = percent * this.audio.duration;
                });

                volumeSlider.addEventListener('input', (e) => {
                    const volume = e.target.value / 100;
                    this.audio.volume = volume;
                    document.getElementById('volumeValue').textContent = `${e.target.value}%`;
                });

                // Zone switching
                document.querySelectorAll('.zone-btn').forEach(btn => {
                    btn.addEventListener('click', (e) => {
                        document.querySelectorAll('.zone-btn').forEach(b => b.classList.remove('active'));
                        e.target.classList.add('active');
                        this.currentZone = e.target.dataset.zone;
                        this.suggestMusicForZone();
                    });
                });

                // Thinking tools
                document.getElementById('thinkBtn').addEventListener('click', () => this.processWithAI());
                document.getElementById('saveBtn').addEventListener('click', () => this.saveEntry());
                document.getElementById('clearBtn').addEventListener('click', () => this.clearJournal());
            }

            setupAudioEvents() {
                this.audio.addEventListener('timeupdate', () => {
                    const progress = (this.audio.currentTime / this.audio.duration) * 100;
                    document.getElementById('progress').style.width = `${progress}%`;
                });

                this.audio.addEventListener('ended', () => {
                    this.playNext();
                });

                this.audio.addEventListener('loadedmetadata', () => {
                    this.updateNowPlaying();
                });
            }

            async handleFileUpload(files) {
                const uploadArea = document.getElementById('uploadArea');
                uploadArea.innerHTML = '<div class="loading"></div> Processing files...';

                for (const file of files) {
                    if (file.type.startsWith('audio/')) {
                        const arrayBuffer = await file.arrayBuffer();
                        const track = {
                            id: crypto.randomUUID(),
                            name: file.name,
                            blob: arrayBuffer,
                            size: file.size,
                            type: file.type,
                            uploaded: new Date().toISOString()
                        };

                        await this.db.add('tracks', track);
                    }
                }

                uploadArea.innerHTML = '🎵 Drop music files here or click to upload';
                this.loadMusicList();
            }

            async loadMusicList() {
                const musicList = document.getElementById('musicList');
                const tracks = await this.db.getAll('tracks');
                
                musicList.innerHTML = '';
                tracks.forEach(track => {
                    const trackElement = document.createElement('div');
                    trackElement.className = 'track-item';
                    trackElement.innerHTML = `
                        <div class="track-info">
                            <div class="track-title">${track.name}</div>
                            <div class="track-artist">Local file</div>
                        </div>
                        <div class="track-duration">${this.formatFileSize(track.size)}</div>
                    `;
                    trackElement.addEventListener('click', () => this.playTrack(track));
                    musicList.appendChild(trackElement);
                });
            }

            async loadPlaylists() {
                const playlists = {
                    focus: await this.db.get('playlists', 'focus') || { tracks: [] },
                    reflection: await this.db.get('playlists', 'reflection') || { tracks: [] },
                    creative: await this.db.get('playlists', 'creative') || { tracks: [] }
                };

                this.renderPlaylist('focusPlaylist', playlists.focus.tracks);
                this.renderPlaylist('reflectionPlaylist', playlists.reflection.tracks);
                this.renderPlaylist('creativePlaylist', playlists.creative.tracks);
            }

            renderPlaylist(elementId, tracks) {
                const element = document.getElementById(elementId);
                element.innerHTML = '';
                tracks.forEach(trackName => {
                    const trackElement = document.createElement('div');
                    trackElement.className = 'playlist-track';
                    trackElement.textContent = trackName;
                    element.appendChild(trackElement);
                });
            }

            async playTrack(track) {
                const blob = new Blob([track.blob], { type: track.type });
                const url = URL.createObjectURL(blob);
                
                this.audio.src = url;
                this.currentTrack = track;
                
                document.querySelectorAll('.track-item').forEach(item => item.classList.remove('playing'));
                event.currentTarget.classList.add('playing');
                
                this.audio.play();
                this.isPlaying = true;
                document.getElementById('playBtn').textContent = '⏸';
            }

            togglePlayPause() {
                if (this.isPlaying) {
                    this.audio.pause();
                    document.getElementById('playBtn').textContent = '▶';
                } else {
                    this.audio.play();
                    document.getElementById('playBtn').textContent = '⏸';
                }
                this.isPlaying = !this.isPlaying;
            }

            updateNowPlaying() {
                const nowPlaying = document.getElementById('nowPlaying');
                if (this.currentTrack) {
                    nowPlaying.innerHTML = `
                        <div class="now-playing-title">${this.currentTrack.name}</div>
                        <div class="now-playing-artist">Local file • ${this.currentZone}</div>
                    `;
                }
            }

            suggestMusicForZone() {
                const zoneSuggestions = {
                    journaling: 'reflection',
                    research: 'focus',
                    problem: 'creative'
                };
                
                const suggestedPlaylist = zoneSuggestions[this.currentZone];
                this.updateNowPlaying();
            }

            async processWithAI() {
                const journalArea = document.querySelector('.journal-area');
                const content = journalArea.value;
                
                if (!content.trim()) {
                    alert('Please write something first!');
                    return;
                }

                // Simulate AI processing
                const thinkBtn = document.getElementById('thinkBtn');
                thinkBtn.innerHTML = '<div class="loading"></div> Thinking...';
                thinkBtn.disabled = true;

                setTimeout(() => {
                    const insights = this.generateInsights(content);
                    journalArea.value += '\n\n--- AI Insights ---\n' + insights;
                    thinkBtn.innerHTML = 'Think with me';
                    thinkBtn.disabled = false;
                }, 2000);
            }

            generateInsights(content) {
                const insights = [];
                
                if (content.length > 100) {
                    insights.push('📝 This is a substantial entry with depth to explore.');
                }
                
                if (content.includes('feel') || content.includes('emotion')) {
                    insights.push('🧘 Emotional awareness detected - consider reflecting on the source.');
                }
                
                if (content.includes('problem') || content.includes('challenge')) {
                    insights.push('💡 Problem-solving mode activated - try breaking this down into smaller steps.');
                }
                
                if (content.includes('idea') || content.includes('think')) {
                    insights.push('🎯 Creative thinking detected - explore connections between concepts.');
                }
                
                return insights.join('\n');
            }

            async saveEntry() {
                const journalArea = document.querySelector('.journal-area');
                const content = journalArea.value;
                
                if (!content.trim()) {
                    alert('Nothing to save!');
                    return;
                }

                const entry = {
                    id: crypto.randomUUID(),
                    content: content,
                    zone: this.currentZone,
                    timestamp: new Date().toISOString(),
                    musicTrack: this.currentTrack ? this.currentTrack.name : null
                };

                await this.db.add('entries', entry);
                
                const saveBtn = document.getElementById('saveBtn');
                saveBtn.textContent = '✓ Saved';
                setTimeout(() => {
                    saveBtn.textContent = 'Save Entry';
                }, 2000);
            }

            clearJournal() {
                if (confirm('Clear the journal area?')) {
                    document.querySelector('.journal-area').value = '';
                }
            }

            playNext() {
                // Simple next track logic
                const tracks = document.querySelectorAll('.track-item');
                const currentIndex = Array.from(tracks).findIndex(track => track.classList.contains('playing'));
                
                if (currentIndex < tracks.length - 1) {
                    tracks[currentIndex + 1].click();
                } else {
                    tracks[0].click();
                }
            }

            formatFileSize(bytes) {
                if (bytes === 0) return '0 Bytes';
                const k = 1024;
                const sizes = ['Bytes', 'KB', 'MB', 'GB'];
                const i = Math.floor(Math.log(bytes) / Math.log(k));
                return parseFloat((bytes / Math.pow(k, i)).toFixed(2)) + ' ' + sizes[i];
            }
        }

        // Initialize the music manager
        const musicManager = new MusicManager();

        // Add IndexedDB polyfill if needed
        if (!window.idb) {
            const script = document.createElement('script');
            script.src = 'https://cdn.jsdelivr.net/npm/idb@7/build/iife/index-min.js';
            document.head.appendChild(script);
        }
    </script>
