/* djzs-xmtp-artifacts.ts
   Copy-paste into your DJZS codebase (TypeScript). This implements:
   - DJZS TradeArtifact (journal + research + thinking partner outputs compiled)
   - Stable hashing (SHA-256) + optional wallet signature
   - XMTP send to a user-run “Trader Agent” + stream back ExecutionReports

   Dependencies:
   - ethers (v6)
   - @xmtp/xmtp-js

   You will also need a connected EVM wallet (for XMTP identity + signing).
*/

import { Client as XmtpClient, type Conversation } from "@xmtp/xmtp-js";
import type { Signer } from "ethers";

/* ----------------------------- Types (DJZS) ----------------------------- */

export type ZoneRef =
  | { kind: "journal"; id: string; title?: string }
  | { kind: "research"; id: string; title?: string }
  | { kind: "url"; url: string; title?: string }
  | { kind: "file"; filename: string; sha256?: string; title?: string };

export interface TradeThesis {
  id: string;
  asset: string; // "TOKEN", "ETH/USDC", etc.
  side: "BUY" | "SELL";
  timeframe: "SHORT" | "MEDIUM" | "LONG";
  conviction: number; // 1..10
  positionSizePct: number; // 0..100
  entry: { type: "MARKET" | "LIMIT"; limitPrice?: number };
  exits?: {
    takeProfitPct?: number;
    stopLossPct?: number;
    invalidation?: string;
  };
  marketConditions?: {
    currentPrice?: number;
    sentiment?: "FEAR" | "NEUTRAL" | "GREED";
    catalysts?: string[];
  };
  notes?: string;
  createdAt: number;
}

export interface StressTestReport {
  approved: boolean; // "AI recommendation", not an execution gate
  reasons: string[];
  counterArguments: string[];
  missingInfo: string[];
  suggestedMemoryPins?: Array<{ title: string; content: string }>;
}

export interface RiskParameters {
  maxPositionSizePct: number; // e.g. 5
  maxDailyLossPct: number; // e.g. 2
  maxAssetExposurePct: number; // e.g. 20 (optional)
  maxSectorExposurePct: number; // e.g. 35 (optional)
}

export interface PortfolioSnapshot {
  timestamp: number;
  totalValueUsd: number;
  realizedPnlTodayUsd: number;
  positions: Array<{
    asset: string;
    valueUsd: number;
    sector?: string;
  }>;
}

export interface RiskSummary {
  withinLimits: boolean;
  violations: string[];
  computed: {
    proposedPositionSizePct: number;
    dailyLossPct: number;
    estWorstCaseLossPct: number;
  };
  params: RiskParameters;
  portfolio?: Pick<PortfolioSnapshot, "timestamp" | "totalValueUsd" | "realizedPnlTodayUsd">;
}

export interface TradeArtifactV1 {
  schema: "djzs.tradeArtifact.v1";
  createdAt: number;

  // Core objects (frozen snapshot)
  thesis: TradeThesis;
  evidence: ZoneRef[];
  stressTest: StressTestReport;
  risk: RiskSummary;

  // Deterministic identity
  hash: string; // sha256 hex of the artifact payload (without signature)
  author: {
    address?: string; // wallet address if available
    signature?: string; // EIP-191 signature of `hash`
    signatureType?: "eip191";
  };

  // Optional: human-friendly rendering for agents/logs
  memo?: string;
}

export interface ExecutionReportV1 {
  schema: "djzs.execReport.v1";
  createdAt: number;
  artifactHash: string;

  status: "RECEIVED" | "REJECTED" | "PAPER_EXECUTED" | "LIVE_SUBMITTED" | "LIVE_CONFIRMED" | "FAILED";
  reason?: string;

  // Optional execution details
  venue?: string;
  details?: Record<string, unknown>;
}

/* -------------------------- Stable stringify + hash -------------------------- */

function isPlainObject(v: unknown): v is Record<string, unknown> {
  return typeof v === "object" && v !== null && !Array.isArray(v);
}

export function stableStringify(value: unknown): string {
  if (value === null) return "null";
  if (typeof value === "string") return JSON.stringify(value);
  if (typeof value === "number" || typeof value === "boolean") return JSON.stringify(value);
  if (Array.isArray(value)) return `[${value.map(stableStringify).join(",")}]`;
  if (isPlainObject(value)) {
    const keys = Object.keys(value).sort();
    const body = keys.map(k => `${JSON.stringify(k)}:${stableStringify(value[k])}`).join(",");
    return `{${body}}`;
  }
  // fallback for undefined / functions / symbols (shouldn't be present)
  return JSON.stringify(String(value));
}

function bytesToHex(bytes: Uint8Array): string {
  return Array.from(bytes).map(b => b.toString(16).padStart(2, "0")).join("");
}

export async function sha256Hex(inputUtf8: string): Promise<string> {
  // Browser + modern runtimes
  const data = new TextEncoder().encode(inputUtf8);
  const digest = await crypto.subtle.digest("SHA-256", data);
  return bytesToHex(new Uint8Array(digest));
}

/* ----------------------------- Risk summary ----------------------------- */

export function computeRiskSummary(args: {
  thesis: TradeThesis;
  params: RiskParameters;
  portfolio?: PortfolioSnapshot;
}): RiskSummary {
  const { thesis, params, portfolio } = args;

  const proposedPositionSizePct = thesis.positionSizePct;
  const dailyLossPct =
    portfolio && portfolio.totalValueUsd > 0
      ? (Math.max(0, -portfolio.realizedPnlTodayUsd) / portfolio.totalValueUsd) * 100
      : 0;

  // Simple deterministic estimate; you can refine later
  const estWorstCaseLossPct =
    typeof thesis.exits?.stopLossPct === "number" ? thesis.exits.stopLossPct : proposedPositionSizePct;

  const violations: string[] = [];
  if (proposedPositionSizePct > params.maxPositionSizePct) violations.push("Position size exceeds maxPositionSizePct");
  if (dailyLossPct > params.maxDailyLossPct) violations.push("Daily loss exceeds maxDailyLossPct");

  return {
    withinLimits: violations.length === 0,
    violations,
    computed: {
      proposedPositionSizePct,
      dailyLossPct,
      estWorstCaseLossPct
    },
    params,
    portfolio: portfolio
      ? {
          timestamp: portfolio.timestamp,
          totalValueUsd: portfolio.totalValueUsd,
          realizedPnlTodayUsd: portfolio.realizedPnlTodayUsd
        }
      : undefined
  };
}

/* ---------------------------- Artifact creation ---------------------------- */

export async function createTradeArtifactV1(args: {
  thesis: TradeThesis;
  evidence: ZoneRef[];
  stressTest: StressTestReport;
  risk: RiskSummary;
  memo?: string;
  signer?: Signer; // optional, but recommended
}): Promise<TradeArtifactV1> {
  const createdAt = Date.now();

  // Payload used for hashing (exclude signature fields so hash is deterministic)
  const payloadForHash = {
    schema: "djzs.tradeArtifact.v1",
    createdAt,
    thesis: args.thesis,
    evidence: args.evidence,
    stressTest: args.stressTest,
    risk: args.risk,
    memo: args.memo ?? undefined
  };

  const hash = await sha256Hex(stableStringify(payloadForHash));

  let address: string | undefined;
  let signature: string | undefined;

  if (args.signer) {
    address = await args.signer.getAddress();
    signature = await args.signer.signMessage(hash); // EIP-191 "personal_sign"
  }

  return {
    schema: "djzs.tradeArtifact.v1",
    createdAt,
    thesis: args.thesis,
    evidence: args.evidence,
    stressTest: args.stressTest,
    risk: args.risk,
    hash,
    author: {
      address,
      signature,
      signatureType: signature ? "eip191" : undefined
    },
    memo: args.memo
  };
}

/* ------------------------------ XMTP bridge ------------------------------ */

export async function createXmtp(signer: Signer): Promise<XmtpClient> {
  // This will create or load the XMTP identity keys for this wallet.
  // In a browser app, you typically persist the XMTP keys yourself; this SDK may prompt and/or store depending on your setup.
  return await XmtpClient.create(signer as any, { env: "production" });
}

async function getConversation(client: XmtpClient, peerAddress: string): Promise<Conversation> {
  return await client.conversations.newConversation(peerAddress);
}

type DjzsEnvelope = { schema: "djzs.envelope.v1"; type: string; body: unknown };

function envelope(type: string, body: unknown): DjzsEnvelope {
  return { schema: "djzs.envelope.v1", type, body };
}

export async function sendTradeArtifactOverXmtp(args: {
  client: XmtpClient;
  agentAddress: string;
  artifact: TradeArtifactV1;
}): Promise<void> {
  const conv = await getConversation(args.client, args.agentAddress);
  const msg = envelope("djzs.tradeArtifact.v1", args.artifact);
  await conv.send(JSON.stringify(msg));
}

export async function streamExecutionReports(args: {
  client: XmtpClient;
  agentAddress: string;
  onReport: (report: ExecutionReportV1, rawMessage: string) => void | Promise<void>;
}): Promise<() => void> {
  const conv = await getConversation(args.client, args.agentAddress);
  const stream = await conv.streamMessages();

  let stopped = false;

  (async () => {
    for await (const m of stream) {
      if (stopped) break;
      const raw = m.content;
      if (typeof raw !== "string") continue;

      try {
        const parsed = JSON.parse(raw) as DjzsEnvelope;
        if (parsed?.schema !== "djzs.envelope.v1") continue;
        if (parsed.type !== "djzs.execReport.v1") continue;

        const report = parsed.body as ExecutionReportV1;
        if (report?.schema !== "djzs.execReport.v1") continue;

        await args.onReport(report, raw);
      } catch {
        // ignore non-JSON
      }
    }
  })();

  return () => {
    stopped = true;
  };
}
