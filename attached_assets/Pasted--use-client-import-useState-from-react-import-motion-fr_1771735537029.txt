"use client";

import { useState } from "react";
import { motion } from "framer-motion";
import { Play, Terminal as TerminalIcon, AlertTriangle, ShieldCheck } from "lucide-react";

// Pre-defined test payloads to simulate A2A logic
const TEST_CASES = {
  fomo: {
    label: "FOMO Momentum Buy",
    memo: "EXECUTE IMMEDIATE BUY: 500 SOL of $SHILL. 1-minute volume is spiking and Crypto Twitter implies a tier-1 exchange listing today. Cannot miss this pump.",
    response: {
      system_id: "djzs-mainnet-01",
      verdict: "FAIL",
      risk_score: 98,
      flags: [{ code: "DJZS-I01", severity: "CRITICAL", description: "FOMO Loop and social momentum dependency detected." }],
      proof: { logic_hash: "0x4a9b2c...", payment_verified: true }
    }
  },
  hallucination: {
    label: "Hallucinated Data",
    memo: "Routing 50k USDC into Yield Protocol V4 based on their latest audit report from yesterday.",
    response: {
      system_id: "djzs-mainnet-01",
      verdict: "FAIL",
      risk_score: 85,
      flags: [{ code: "DJZS-E01", severity: "HIGH", description: "Epistemic Failure. Yield Protocol V4 does not exist and no audit was published." }],
      proof: { logic_hash: "0x7f2e1a...", payment_verified: true }
    }
  },
  valid: {
    label: "Valid Strategy",
    memo: "Executing DCA of 2 ETH. Structural support verified at $2800. Liquidity depth is sufficient. Max slippage set to 0.5%.",
    response: {
      system_id: "djzs-mainnet-01",
      verdict: "PASS",
      risk_score: 12,
      flags: [],
      proof: { logic_hash: "0x9b3a4f...", payment_verified: true }
    }
  }
};

export default function InteractiveTerminal() {
  const [activeCase, setActiveCase] = useState<keyof typeof TEST_CASES>("fomo");
  const [isScanning, setIsScanning] = useState(false);
  const [output, setOutput] = useState<any>(null);

  const runAudit = () => {
    setIsScanning(true);
    setOutput(null);
    // Simulate network latency and Oracle processing time
    setTimeout(() => {
      setIsScanning(false);
      setOutput(TEST_CASES[activeCase].response);
    }, 1200);
  };

  return (
    <section id="demo" className="py-24 bg-[#0a0a0a] text-gray-200 border-t border-gray-900 font-sans">
      <div className="max-w-6xl mx-auto px-6 lg:px-8">
        
        <div className="text-center mb-16">
          <h2 className="text-3xl lg:text-4xl font-bold tracking-tight mb-4 text-white">
            Test the Oracle
          </h2>
          <p className="text-gray-500 max-w-xl mx-auto">
            Inject a raw reasoning trace and watch the Adversarial Logic Layer parse, grade, and cryptographically verify the payload.
          </p>
        </div>

        <div className="grid grid-cols-1 lg:grid-cols-2 gap-8 bg-[#111] border border-gray-800 rounded-2xl overflow-hidden shadow-2xl">
          
          {/* Left Panel: Input & Controls */}
          <div className="p-8 border-b lg:border-b-0 lg:border-r border-gray-800 bg-[#0d0d0d]">
            <div className="flex items-center space-x-2 mb-6">
              <TerminalIcon size={20} className="text-cyan-400" />
              <h3 className="font-semibold text-white">Agent Payload Injector</h3>
            </div>

            <div className="space-y-3 mb-8">
              {(Object.keys(TEST_CASES) as Array<keyof typeof TEST_CASES>).map((key) => (
                <button
                  key={key}
                  onClick={() => { setActiveCase(key); setOutput(null); }}
                  className={`w-full text-left px-4 py-3 rounded-lg border transition-all ${
                    activeCase === key 
                    ? "border-purple-500 bg-purple-500/10 text-purple-300" 
                    : "border-gray-800 hover:border-gray-600 text-gray-400"
                  }`}
                >
                  {TEST_CASES[key].label}
                </button>
              ))}
            </div>

            <div className="mb-8">
              <div className="text-xs font-mono text-gray-500 mb-2">"strategy_memo":</div>
              <div className="p-4 rounded-lg bg-black border border-gray-800 font-mono text-sm text-gray-300 h-32 overflow-y-auto">
                {TEST_CASES[activeCase].memo}
              </div>
            </div>

            <button
              onClick={runAudit}
              disabled={isScanning}
              className="w-full flex items-center justify-center space-x-2 py-4 rounded-lg bg-white text-black font-bold hover:bg-gray-200 disabled:opacity-50 transition-colors"
            >
              {isScanning ? (
                <motion.div animate={{ rotate: 360 }} transition={{ repeat: Infinity, duration: 1, ease: "linear" }}>
                  <TerminalIcon size={20} />
                </motion.div>
              ) : (
                <Play size={20} />
              )}
              <span>{isScanning ? "ORACLE SCANNING..." : "EXECUTE JET AUDIT"}</span>
            </button>
          </div>

          {/* Right Panel: Output Terminal */}
          <div className="p-8 bg-[#050505] relative min-h-[400px]">
            <div className="text-xs font-mono text-gray-500 mb-4 border-b border-gray-800 pb-2">ProofOfLogic Response</div>
            
            {isScanning && (
              <div className="flex flex-col items-center justify-center h-full space-y-4 opacity-70">
                <div className="w-full h-1 bg-gray-900 rounded-full overflow-hidden">
                  <motion.div 
                    className="h-full bg-cyan-500 shadow-[0_0_10px_#06b6d4]"
                    initial={{ width: "0%" }}
                    animate={{ width: "100%" }}
                    transition={{ duration: 1.2, ease: "linear" }}
                  />
                </div>
                <div className="font-mono text-xs text-cyan-400 animate-pulse">ANALYZING DJZS-LF TAXONOMY...</div>
              </div>
            )}

            {!isScanning && output && (
              <motion.div 
                initial={{ opacity: 0, y: 10 }}
                animate={{ opacity: 1, y: 0 }}
                className="font-mono text-sm"
              >
                <div className="mb-4 flex items-center space-x-2">
                  {output.verdict === "FAIL" ? (
                    <span className="px-2 py-1 bg-red-500/20 text-red-400 rounded border border-red-500/30 flex items-center text-xs">
                      <AlertTriangle size={14} className="mr-1" /> VERDICT: FAIL
                    </span>
                  ) : (
                    <span className="px-2 py-1 bg-green-500/20 text-green-400 rounded border border-green-500/30 flex items-center text-xs">
                      <ShieldCheck size={14} className="mr-1" /> VERDICT: PASS
                    </span>
                  )}
                  <span className="px-2 py-1 bg-gray-900 text-gray-400 rounded border border-gray-800 text-xs">
                    RISK: {output.risk_score}/100
                  </span>
                </div>
                <pre className="text-gray-300 overflow-x-auto">
                  <code>{JSON.stringify(output, null, 2)}</code>
                </pre>
              </motion.div>
            )}

            {!isScanning && !output && (
              <div className="flex items-center justify-center h-full font-mono text-gray-600 text-sm">
                Awaiting payload injection...
              </div>
            )}
          </div>

        </div>
      </div>
    </section>
  );
}
