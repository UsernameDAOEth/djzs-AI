### Step 2: Create the XMTP Wagmi Adapter (The Hook)

XMTP requires a `Signer` interface, but Wagmi provides a `WalletClient`. We need to create a custom hook to translate Wagmi's hex signatures into the `Uint8Array` format that XMTP expects.

Create a new file: `client/src/hooks/useXmtp.ts`

```typescript
import { useState, useCallback } from "react";
import { Client, IdentifierKind, Signer } from "@xmtp/browser-sdk";
import { useWalletClient } from "wagmi";
import { toBytes } from "viem";

export function useXmtp() {
  const { data: walletClient } = useWalletClient();
  const [client, setClient] = useState<Client | null>(null);
  const [isInitializing, setIsInitializing] = useState(false);

  const connectXmtp = useCallback(async () => {
    if (!walletClient) throw new Error("Wallet not connected");
    setIsInitializing(true);

    try {
      // 1. Create the XMTP Signer adapter for Wagmi
      const signer: Signer = {
        type: "EOA",
        getIdentifier: () => ({
          identifier: walletClient.account.address,
          identifierKind: IdentifierKind.Ethereum,
        }),
        signMessage: async (message: string) => {
          // Wagmi signs the message, viem converts the hex to Uint8Array
          const signature = await walletClient.signMessage({ message });
          return toBytes(signature);
        },
      };

      // 2. Initialize the client (Using 'production' for real founders)
      const xmtpClient = await Client.create(signer, { env: "production" });
      setClient(xmtpClient);
      return xmtpClient;

    } catch (error) {
      console.error("Failed to connect to XMTP:", error);
      throw error;
    } finally {
      setIsInitializing(false);
    }
  }, [walletClient]);

  return { client, connectXmtp, isInitializing };
}

```

### Step 3: The "Founder Sparring Room" UI Component

This is the actual chat interface. It takes the target founder's wallet address, initializes the secure connection, and streams messages in real-time.

Create a new file: `client/src/components/SparringRoom.tsx`

```tsx
import { useState, useEffect, useRef } from "react";
import { IdentifierKind } from "@xmtp/browser-sdk";
import { useXmtp } from "../hooks/useXmtp";
import { Send, Shield, Lock } from "lucide-react";

interface SparringRoomProps {
  founderAddress: string; // The 0x address of your advisory client
}

export function SparringRoom({ founderAddress }: SparringRoomProps) {
  const { client, connectXmtp, isInitializing } = useXmtp();
  const [conversation, setConversation] = useState<any>(null);
  const [messages, setMessages] = useState<any[]>([]);
  const [inputValue, setInputValue] = useState("");
  
  const messagesEndRef = useRef<HTMLDivElement>(null);

  // Initialize Chat & Stream
  useEffect(() => {
    const startChat = async () => {
      if (!client) return;
      
      try {
        // Create a secure 1-on-1 channel using their Ethereum address
        const group = await client.conversations.newGroupWithIdentities([
          { identifier: founderAddress, identifierKind: IdentifierKind.Ethereum }
        ]);
        setConversation(group);

        // Load historical messages
        const history = await group.messages();
        setMessages(history);

        // Stream new incoming messages
        await group.streamMessages({
          onValue: (msg: any) => {
            setMessages((prev) => [...prev, msg]);
          }
        });
      } catch (error) {
        console.error("Error setting up conversation:", error);
      }
    };

    if (client) startChat();
  }, [client, founderAddress]);

  // Auto-scroll to bottom
  useEffect(() => {
    messagesEndRef.current?.scrollIntoView({ behavior: "smooth" });
  }, [messages]);

  const handleSend = async () => {
    if (!inputValue.trim() || !conversation) return;
    try {
      // Send the message to the XMTP network
      await conversation.sendText(inputValue);
      setInputValue("");
    } catch (err) {
      console.error("Failed to send message", err);
    }
  };

  // If not connected to XMTP yet, show the "Unlock Vault" screen
  if (!client) {
    return (
      <div className="flex flex-col items-center justify-center h-96 border border-zinc-800 bg-black text-white rounded-lg p-8">
        <Shield className="w-12 h-12 mb-4 text-zinc-500" />
        <h2 className="text-xl font-bold mb-2">Confidential Sparring Room</h2>
        <p className="text-zinc-400 mb-6 text-center">
          End-to-end encrypted logic audits. Sign to unlock the vault.
        </p>
        <button
          onClick={connectXmtp}
          disabled={isInitializing}
          className="bg-white text-black px-6 py-2 rounded-md font-medium hover:bg-zinc-200 transition-colors"
        >
          {isInitializing ? "Decrypting..." : "Initialize XMTP"}
        </button>
      </div>
    );
  }

  // The Active Chat Interface
  return (
    <div className="flex flex-col h-[600px] border border-zinc-800 bg-black text-white rounded-lg">
      {/* Header */}
      <div className="p-4 border-b border-zinc-800 flex items-center justify-between">
        <div>
          <h3 className="font-bold">Logic Audit Session</h3>
          <p className="text-xs text-zinc-500">{founderAddress}</p>
        </div>
        <Lock className="w-4 h-4 text-zinc-500" />
      </div>

      {/* Messages Area */}
      <div className="flex-1 overflow-y-auto p-4 space-y-4">
        {messages.map((msg, idx) => {
          const isMe = msg.senderAddress.toLowerCase() === client.accountAddress.toLowerCase();
          return (
            <div key={idx} className={`flex ${isMe ? "justify-end" : "justify-start"}`}>
              <div className={`max-w-[80%] p-3 rounded-lg ${
                isMe ? "bg-zinc-800 text-white" : "bg-zinc-900 border border-zinc-800 text-zinc-300"
              }`}>
                <p className="text-sm">{msg.content}</p>
              </div>
            </div>
          );
        })}
        <div ref={messagesEndRef} />
      </div>

      {/* Input Area */}
      <div className="p-4 border-t border-zinc-800">
        <div className="flex space-x-2">
          <input
            type="text"
            value={inputValue}
            onChange={(e) => setInputValue(e.target.value)}
            onKeyPress={(e) => e.key === "Enter" && handleSend()}
            placeholder="Interrogate the logic..."
            className="flex-1 bg-zinc-900 border border-zinc-800 rounded-md px-4 py-2 text-sm focus:outline-none focus:border-zinc-500"
          />
          <button
            onClick={handleSend}
            className="bg-white text-black p-2 rounded-md hover:bg-zinc-200 transition-colors"
          >
            <Send className="w-5 h-5" />
          </button>
        </div>
      </div>
    </div>
  );
}

```
