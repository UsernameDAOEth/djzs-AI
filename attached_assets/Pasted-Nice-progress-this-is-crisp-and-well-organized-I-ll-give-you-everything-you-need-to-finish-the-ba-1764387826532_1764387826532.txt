// Full working route with streamifier
import express from "express";
import multer from "multer";
import PinataSDK from "@pinata/sdk";
import streamifier from "streamifier";
import dotenv from "dotenv";

dotenv.config();
const router = express.Router();

const storage = multer.memoryStorage();
const upload = multer({
  storage,
  limits: { fileSize: 20 * 1024 * 1024 },
});

const PINATA_API_KEY = process.env.PINATA_API_KEY!;
const PINATA_SECRET = process.env.PINATA_SECRET!;
const pinata = PinataSDK(PINATA_API_KEY, PINATA_SECRET);

router.post("/api/journal/upload", upload.single("file"), async (req, res) => {
  try {
    if (!req.file) return res.status(400).json({ success: false, error: "No file" });

    const { originalname, buffer, mimetype } = req.file;

    const fileStream = streamifier.createReadStream(buffer);

    const options = {
      pinataMetadata: { name: originalname },
      pinataOptions: { cidVersion: 1 },
    };

    const pinResponse = await pinata.pinFileToIPFS(fileStream, options);
    // pinResponse example: { IpfsHash: 'Qm...', PinSize: 12345, Timestamp: '...' }

    // Build IPFS URL (use gateway or your preference)
    const ipfsHash = pinResponse.IpfsHash;

3) Upload flow & metadata (recommended)

Typical flow for minting an NFT:
	1.	Client uploads file to /api/journal/upload via multipart/form-data field file.
	2.	Backend pins file and returns ipfsHash (e.g., Qm...).
	3.	Backend (or client) pins metadata JSON to Pinata that references the file: image: "ipfs://<fileHash>"
	4.	Mint contract is called with tokenURI = "ipfs://<metadataHash>"

I recommend pinning metadata from the backend (so metadata IPFS hash is consistent and you can validate fields before minting), but pinning metadata from the client is also possible.

Example metadata JSON:

{
  "name": "My Journal Entry Title",
  "description": "Private/public journal minted on Base/Aztec",
  "image": "ipfs://QmFileHash",
  "attributes": [
    { "trait_type": "visibility", "value": "private" }
  ]
}

Pin metadata using Pinata SDK:

const metadata = {
  name: originalname,
  description: `Journal entry by ${uploader || 'unknown'}`,
  image: `ipfs://${ipfsHash}`,
};

const metaRes = await pinata.pinJSONToIPFS(metadata, {
  pinataMetadata: { name: `${originalname}-metadata` },
});
const metadataIpfsHash = metaRes.IpfsHash; // use ipfs://metadataIpfsHash as tokenURI

Return both file ipfsHash and metadata ipfsHash so client can use tokenURI = ipfs://<metadataIpfsHash> when calling the contract.

4) Frontend — accept uploadedFile in mint buttons

Assuming your frontend already captures files and passes them to the mint buttons. Here are minimal TypeScript/React edits for both buttons.

Props interface

// types.ts or component file
export interface UploadedFile {
  name: string;
  size: number;
  type: string;
  // either a File object (during client-side) or server result:
  file?: File;
  ipfsHash?: string;            // after backend upload (Qm...)
  metadataIpfsHash?: string;    // after metadata pin
  ipfsUrl?: string;             // https://ipfs.io/ipfs/Qm...
}

PrivacyMintButton.tsx

import React from "react";

interface Props {
  uploadedFile?: UploadedFile;
  onMint?: (txHash: string) => void;
  // other props you already have
}

export const PrivacyMintButton: React.FC<Props> = ({ uploadedFile, onMint, ...rest }) => {
  const [loading, setLoading] = React.useState(false);

  const handleMint = async () => {
    try {
      setLoading(true);

      let tokenURI: string | undefined;

      // If backend already returned metadataIpfsHash, use it
      if (uploadedFile?.metadataIpfsHash) {
        tokenURI = `ipfs://${uploadedFile.metadataIpfsHash}`;
      } else if (uploadedFile?.file && !uploadedFile.ipfsHash) {
        // upload the file to backend then get metadata and use it
        const form = new FormData();
        form.append("file", uploadedFile.file);

        const uploadRes = await fetch("/api/journal/upload", {
          method: "POST",
          body: form,
        });
        const uploadJson = await uploadRes.json();
        if (!uploadJson.success) throw new Error(uploadJson.error || "Upload failed");

        // pin metadata via backend (optionally have endpoint to pin metadata)
        const metadataRes = await fetch("/api/journal/metadata", {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({
            name: uploadedFile.name,
            description: "Private journal entry",
            image: `ipfs://${uploadJson.ipfsHash}`,
            visibility: "private",
          }),
        });
        const metaJson = await metadataRes.json();
        if (!metaJson.success) throw new Error(metaJson.error || "Metadata pin failed");

        tokenURI = `ipfs://${metaJson.metadataIpfsHash}`;
      } else if (uploadedFile?.ipfsHash) {
        // if file pinned but metadata not
        // create metadata and pin (client or backend)
        const metaRes = await fetch("/api/journal/metadata", {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({
            name: uploadedFile.name,
            description: "Private journal entry",
            image: `ipfs://${uploadedFile.ipfsHash}`,
            visibility: "private",
          }),
        });
        const metaJson = await metaRes.json();
        if (!metaJson.success) throw new Error(metaJson.error || "Metadata pin failed");
        tokenURI = `ipfs://${metaJson.metadataIpfsHash}`;
      } else {
        // no uploaded file — fallback: mint without image or show error
        throw new Error("No uploaded file provided for minting");
      }

      // Now call your privacy mint function with tokenURI
      // Example: await privacyContract.mintToken(toAddress, tokenURI);
      // This depends on your existing mint code.
      const txHash = "0x..."; // call your contract and get tx hash
      onMint?.(txHash);

    } catch (err: any) {
      console.error("Mint failed:", err);
      alert("Mint failed: " + (err.message || err));
    } finally {
      setLoading(false);
    }
  };

  return (
    <button onClick={handleMint} disabled={loading}>
      {loading ? "Minting..." : "Privacy Mint"}
    </button>
  );
};

MintButton.tsx (public mint) — same pattern but set visibility: "public" in metadata and call your public mint contract.

Important: If your minting is done entirely client-side (user signs transaction and calls contract), ensure tokenURI points to ipfs://<metadataHash> before the transaction submission. You can either:
	•	Have backend return metadataIpfsHash after pinning metadata (recommended), or
	•	Pin metadata from the client directly using a Pinata client (but that exposes API keys — do not expose Pinata secret key to client).

So: best practice — client uploads file → backend pins file → backend pins metadata → backend returns metadataIpfsHash → client calls contract with tokenURI=ipfs://metadataIpfsHash.

5) Add /api/journal/metadata backend endpoint (optional but recommended)

If you want backend to pin metadata:

router.post("/api/journal/metadata", express.json(), async (req, res) => {
  try {
    const { name, description, image, attributes } = req.body;
    if (!image) return res.status(400).json({ success: false, error: "image required" });

    const metadata = { name, description, image, attributes: attributes || [] };
    const metaRes = await pinata.pinJSONToIPFS(metadata, {
      pinataMetadata: { name: `${name || "journal-entry"}-metadata` },
    });

    return res.json({ success: true, metadataIpfsHash: metaRes.IpfsHash, metadataUrl: `https://ipfs.io/ipfs/${metaRes.IpfsHash}` });
  } catch (err: any) {
    console.error("pin metadata error", err);
    return res.status(500).json({ success: false, error: err.message || String(err) });
  }
});

7) Tips / caveats / security
	•	Pinata keys must never be shipped to client. Keep PINATA_SECRET only on backend.
	•	File size: set multer limits and reject too-large files. Consider streaming to a cloud storage (S3) if huge files are expected, then pin smaller derived assets to IPFS.
	•	IPFS gateway: Using ipfs:// is best for tokenURI. For user previews, you can use https://ipfs.io/ipfs/<hash> or your preferred gateway.
	•	Pin persistence: Pinata pins but if you rely on persistence, consider backup pinning or use an IPFS cluster or a third-party persistent pin service.
	•	Private (Aztec) vs Public (Base): For Aztec (private chain), you may want to avoid placing metadata content exposing private data — instead pin encrypted content and publish only encrypted key/metadata publicly. If you need help adding encryption before pinning, I can provide a simple client-side AES encryption flow so the file on IPFS is encrypted and the metadata references an encrypted file — then include decryption key in private contract storage or provide to intended recipient only.
	•	Hardhat peer deps: the --legacy-peer-deps solution is usually enough. If hardhat still breaks, temporarily install multer in a small side-service, or containerize the backend.
