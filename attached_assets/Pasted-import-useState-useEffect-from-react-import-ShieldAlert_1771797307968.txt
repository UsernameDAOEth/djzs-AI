import { useState, useEffect } from "react";
import { ShieldAlert, ShieldCheck, Activity, Database, ArrowUpRight, Loader2 } from "lucide-react";

interface AuditLog {
  id: string;
  agent: string;
  payload: string;
  verdict: "PASS" | "FAIL";
  code: string;
  risk: number;
  time: string;
}

export default function DashboardOverview() {
  const [audits, setAudits] = useState<AuditLog[]>([]);
  const [isLoading, setIsLoading] = useState(true);

  // Fetch live data from your backend
  useEffect(() => {
    const fetchLogs = async () => {
      try {
        const response = await fetch('/api/audit/logs');
        if (!response.ok) throw new Error("Failed to fetch logs");
        const data = await response.json();
        setAudits(data);
      } catch (error) {
        console.error("Error loading console data:", error);
      } finally {
        setIsLoading(false);
      }
    };

    fetchLogs();
    
    // Optional: Set up a polling interval to refresh data every 10 seconds
    const interval = setInterval(fetchLogs, 10000);
    return () => clearInterval(interval);
  }, []);

  // Dynamically calculate telemetry
  const totalAudits = audits.length;
  const threatsPrevented = audits.filter(a => a.verdict === "FAIL").length;
  const avgRisk = totalAudits > 0 
    ? Math.round(audits.reduce((acc, curr) => acc + curr.risk, 0) / totalAudits) 
    : 0;

  if (isLoading) {
    return (
      <div className="flex flex-col items-center justify-center h-full text-cyan-400 space-y-4">
        <Loader2 size={32} className="animate-spin" />
        <div className="text-sm font-mono tracking-widest">SYNCING WITH DATABASE...</div>
      </div>
    );
  }

  return (
    <div className="max-w-6xl mx-auto space-y-8 animate-in fade-in duration-500">
      
      {/* Page Header */}
      <div>
        <h1 className="text-2xl font-bold text-white mb-1">Live Terminal</h1>
        <p className="text-sm text-gray-500">Real-time logic monitoring for your autonomous agents.</p>
      </div>

      {/* Dynamic Telemetry Row */}
      <div className="grid grid-cols-1 md:grid-cols-3 gap-6">
        <div className="p-6 rounded-xl bg-[#0a0a0a] border border-gray-800 shadow-lg relative overflow-hidden group hover:border-cyan-500/50 transition-colors">
          <div className="absolute top-0 right-0 p-4 opacity-5 group-hover:opacity-10 transition-opacity"><Database size={64} className="text-cyan-400" /></div>
          <div className="text-xs font-mono text-gray-500 mb-2">TOTAL AUDITS (RECORDED)</div>
          <div className="text-3xl font-bold text-white">{totalAudits}</div>
          <div className="text-xs text-cyan-400 mt-2 flex items-center"><ArrowUpRight size={12} className="mr-1"/> Live Sync Active</div>
        </div>
        
        <div className="p-6 rounded-xl bg-[#0a0a0a] border border-gray-800 shadow-lg relative overflow-hidden group hover:border-red-500/50 transition-colors">
          <div className="absolute top-0 right-0 p-4 opacity-5 group-hover:opacity-10 transition-opacity"><ShieldAlert size={64} className="text-red-400" /></div>
          <div className="text-xs font-mono text-gray-500 mb-2">THREATS PREVENTED</div>
          <div className="text-3xl font-bold text-white">{threatsPrevented}</div>
          <div className="text-xs text-red-400 mt-2">Capital destruction halted</div>
        </div>

        <div className="p-6 rounded-xl bg-[#0a0a0a] border border-gray-800 shadow-lg relative overflow-hidden group hover:border-purple-500/50 transition-colors">
          <div className="absolute top-0 right-0 p-4 opacity-5 group-hover:opacity-10 transition-opacity"><Activity size={64} className="text-purple-400" /></div>
          <div className="text-xs font-mono text-gray-500 mb-2">AVG RISK SCORE</div>
          <div className="text-3xl font-bold text-white">{avgRisk}<span className="text-gray-600 text-lg">/100</span></div>
          <div className="text-xs text-purple-400 mt-2">Aggregated from ledger</div>
        </div>
      </div>

      {/* The Live Audit Ledger */}
      <div className="bg-[#0a0a0a] border border-gray-800 rounded-xl overflow-hidden shadow-2xl">
        <div className="px-6 py-4 border-b border-gray-800 flex justify-between items-center bg-[#111]">
          <h2 className="text-xs font-bold text-gray-400 uppercase tracking-wider font-mono flex items-center">
            <span className="w-2 h-2 rounded-full bg-green-500 animate-pulse mr-2" />
            Live Executions
          </h2>
        </div>
        
        <div className="overflow-x-auto">
          <table className="w-full text-left text-sm font-mono whitespace-nowrap">
            <thead className="bg-[#050505] text-gray-500 text-xs border-b border-gray-900">
              <tr>
                <th className="px-6 py-4 font-medium">Request ID</th>
                <th className="px-6 py-4 font-medium">Agent Group</th>
                <th className="px-6 py-4 font-medium">Payload Preview</th>
                <th className="px-6 py-4 font-medium">Risk</th>
                <th className="px-6 py-4 font-medium">Verdict</th>
                <th className="px-6 py-4 font-medium text-right">Time</th>
              </tr>
            </thead>
            <tbody className="divide-y divide-gray-900 text-gray-300">
              {audits.length === 0 ? (
                <tr>
                  <td colSpan={6} className="px-6 py-8 text-center text-gray-600">Awaiting agent payloads...</td>
                </tr>
              ) : (
                audits.map((audit) => (
                  <tr key={audit.id} className="hover:bg-[#111] transition-colors group cursor-pointer">
                    <td className="px-6 py-4 text-gray-500 group-hover:text-cyan-400 transition-colors">{audit.id}</td>
                    <td className="px-6 py-4">{audit.agent}</td>
                    <td className="px-6 py-4 truncate max-w-[200px] text-gray-400">{audit.payload}</td>
                    <td className="px-6 py-4">
                      <span className={`px-2 py-1 rounded text-xs border ${audit.risk > 80 ? 'border-red-500/30 text-red-400 bg-red-500/10' : 'border-gray-800 text-gray-400 bg-gray-900'}`}>
                        {audit.risk}
                      </span>
                    </td>
                    <td className="px-6 py-4">
                      {audit.verdict === "PASS" ? (
                        <span className="flex items-center text-green-400 text-xs"><ShieldCheck size={14} className="mr-1" /> PASS</span>
                      ) : (
                        <span className="flex items-center text-red-400 text-xs"><ShieldAlert size={14} className="mr-1" /> {audit.code}</span>
                      )}
                    </td>
                    <td className="px-6 py-4 text-right text-gray-600 text-xs">{audit.time}</td>
                  </tr>
                ))
              )}
            </tbody>
          </table>
        </div>
      </div>
    </div>
  );
}
